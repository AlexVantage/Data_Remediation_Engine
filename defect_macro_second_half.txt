Sub InjectDefects()

    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("01_RAW_EXPORT")
    
    Dim lastRow As Long
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    
    Dim i As Long
    Dim rand As Double
    
    ' Seed randomizer
    Randomize
    
    For i = 2 To lastRow
        rand = Rnd()
        
        ' --- SKU (Col F): Remove hyphens from ~20% of rows ---
        If rand < 0.2 Then
            ws.Cells(i, 6).Value = Replace(ws.Cells(i, 6).Value, "-", "")
        End If
        
        ' --- Sales_Channel (Col H): Inject invalid value in ~10% of rows ---
        If Rnd() < 0.1 Then
            Dim invalidChannel(2) As String
            invalidChannel(0) = "In-Store"
            invalidChannel(1) = "Ordered"
            invalidChannel(2) = "Direct"
            ws.Cells(i, 8).Value = invalidChannel(Int(Rnd() * 3))
        End If
        
        ' --- Fulfillment_Status (Col I): Inject invalid value in ~10% of rows ---
        If Rnd() < 0.1 Then
            Dim invalidStatus(2) As String
            invalidStatus(0) = "Processing"
            invalidStatus(1) = "Cancelled"
            invalidStatus(2) = "Unknown"
            ws.Cells(i, 9).Value = invalidStatus(Int(Rnd() * 3))
        End If
        
        ' --- Revenue_USD (Col J): Store as text string in ~15% of rows ---
        If Rnd() < 0.15 Then
            Dim revVal As String
            revVal = CStr(ws.Cells(i, 10).Value)
            ws.Cells(i, 10).NumberFormat = "@"
            ws.Cells(i, 10).Value = revVal
        End If
        
        ' --- Quantity (Col K): Store as text string in ~15% of rows ---
        If Rnd() < 0.15 Then
            Dim qtyVal As String
            qtyVal = CStr(ws.Cells(i, 11).Value)
            ws.Cells(i, 11).NumberFormat = "@"
            ws.Cells(i, 11).Value = qtyVal
        End If
        
    Next i
    
    MsgBox "Defects injected into 01_RAW_EXPORT. Rows processed: " & (lastRow - 1)

End Sub
